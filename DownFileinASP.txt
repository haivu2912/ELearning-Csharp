private string GetFileTypeExtension(string fileExtension)
{
	switch(fileExtension.ToLower())
	{
		case ".docx":
		case ".doc":
			return "MS Word Document"
		case ".xlsx":
		case ".xls":
			return "MS Excel Document"
		case ".txt":
			return "Text Document";
		default:
			return "unknow";
	}
}
